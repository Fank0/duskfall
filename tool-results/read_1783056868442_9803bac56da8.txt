   194→async function narrateAction(
   195→  roomCode: string,
   196→  actorName: string,
   197→  playerAction: string,
   198→  data: {
   199→    playerRolls: ResolvedRoll[];
   200→    outcome: "success" | "failure";
   201→    branchNarrative: string;
   202→    damageToMonster: number;
   203→    monsterThatDied: string | null;
   204→    inventoryChanges: InventoryChange[];
   205→    goldChange: number;
   206→    location: string;
   207→  }
   208→): Promise<string> {
   209→  const lines: string[] = [];
   210→  lines.push(`Локация: ${data.location}`);
   211→  lines.push(`Герой: ${actorName}`);
   212→  lines.push(`Действие: ${playerAction}`);
   213→  lines.push(`Исход: ${data.outcome === "success" ? "УСПЕХ" : "ПРОВАЛ"}`);
   214→  for (const r of data.playerRolls) {
   215→    lines.push(
   216→      `- ${r.label}: ${r.notation}${r.modifier >= 0 ? "+" : ""}${r.modifier} = ${r.total} (выпало ${r.result})${r.target ? `, цель ${r.target}` : ""} → ${r.success ? "успех" : "провал"}`
   217→    );
   218→  }
   219→  if (data.damageToMonster > 0) lines.push(`Урон противнику: ${data.damageToMonster}`);
   220→  if (data.monsterThatDied) lines.push(`Повержен: ${data.monsterThatDied}`);
   221→  if (data.inventoryChanges.length > 0)
   222→    lines.push("Изменения инвентаря: " + data.inventoryChanges.map((c) => `${c.action === "add" ? "+" : "-"}${c.item}`).join(", "));
   223→  if (data.goldChange) lines.push(`Золото: ${data.goldChange > 0 ? "+" : ""}${data.goldChange}`);
   224→  lines.push(`Заготовка нарратива: ${data.branchNarrative}`);
   225→
   226→  try {
   227→    const text = await chatComplete([
   228→      { role: "system", content: SYSTEM_PROMPT_NARRATION },
   229→      { role: "user", content: `Напиши повествование:\n${lines.join("\n")}` },
   230→    ]);
   231→    if (text && text.trim().length > 20) return text.trim();
   232→  } catch (e) {
   233→    console.error("[DM] narrateAction error:", e);
   234→  }
   235→  return data.branchNarrative;
   236→}

   239→export async function* streamNarrativeAction(
   240→  roomCode: string,
   241→  actorName: string,
   242→  playerAction: string,
   243→  data: {
   244→    playerRolls: ResolvedRoll[];
   245→    outcome: "success" | "failure";
   246→    branchNarrative: string;
   247→    damageToMonster: number;
   248→    monsterThatDied: string | null;
   249→    inventoryChanges: InventoryChange[];
   250→    goldChange: number;
   251→    location: string;
   252→  }
   253→): AsyncGenerator<string> {
   254→  const lines: string[] = [];
   255→  lines.push(`Локация: ${data.location}`);
   256→  lines.push(`Герой: ${actorName}`);
   257→  lines.push(`Действие: ${playerAction}`);
   258→  lines.push(`Исход: ${data.outcome === "success" ? "УСПЕХ" : "ПРОВАЛ"}`);
   259→  for (const r of data.playerRolls) {
   260→    lines.push(`- ${r.label}: ${r.notation}${r.modifier >= 0 ? "+" : ""}${r.modifier} = ${r.total} → ${r.success ? "успех" : "провал"}`);
   261→  }
   262→  if (data.damageToMonster > 0) lines.push(`Урон противнику: ${data.damageToMonster}`);
   263→  if (data.monsterThatDied) lines.push(`Повержен: ${data.monsterThatDied}`);
   264→  if (data.goldChange) lines.push(`Золото: ${data.goldChange > 0 ? "+" : ""}${data.goldChange}`);
   265→  lines.push(`Заготовка: ${data.branchNarrative}`);
   266→
   267→  try {
   268→    let full = "";
   269→    for await (const delta of chatStream([
   270→      { role: "system", content: SYSTEM_PROMPT_NARRATION },
   271→      { role: "user", content: `Напиши повествование (3-5 предложений):\n${lines.join("\n")}` },
   272→    ])) {
   273→      full += delta;
   274→      yield delta;
   275→    }
   276→    if (full.trim().length > 20) return;
   277→  } catch (e) {
   278→    console.error("[DM] streamNarrativeAction error:", e);
   279→  }
   280→  // Fallback: yield the branch narrative as a single chunk.
   281→  yield data.branchNarrative;
   282→}

   285→  roomCode: string,
   286→  data: {
   287→    monsterName: string;
   288→    moved: boolean;
   289→    targetName: string | null;
   290→    hit: boolean | null;

   292→    attackTotal: number | null;

   294→    location: string;
   295→  }

   305→    lines.push(`Атака попадает по ${data.targetName}: бросок ${data.attackTotal} против AC ${data.ac}, урон ${data.damage}.`);
   306→  } else {
   307→    lines.push(`Атака промахивается по ${data.targetName}: бросок ${data.attackTotal} против AC ${data.ac}.`);

   310→    const text = await chatComplete([

   312→      { role: "user", content: `Напиши короткое повествование (2-4 предложения) хода монстра:\n${lines.join("\n")}` },
   313→    ]);
   314→    if (text && text.trim().length > 15) return text.trim();
   315→  } catch (e) {
   316→    console.error("[DM] narrateMonsterTurn error:", e);
   317→  }
   318→  // fallback
   319→  if (data.moved) return `${data.monsterName} с рыком бросается вперёд, сокращая дистанцию до ${data.targetName}.`;
   320→  if (data.hit) return `${data.monsterName} бьёт ${data.targetName} и попадает — ${data.damage} урона!`;
   321→  if (data.hit === false) return `${data.monsterName} замахивается по ${data.targetName}, но промахивается.`;
   322→  return `${data.monsterName} маневрирует на поле боя.`;
   323→}

   343→async function resolvePlayerAction(

   347→  playerAction: string,
   348→  round: number,
   349→  plan: DMResolution

   351→  // Fetch the acting player's full state (for talent modifiers).
   352→  const snap0 = await getSnapshot(roomCode);
   353→  const actorState: PlayerState | undefined = snap0?.players.find((p) => p.name === actorName);

   357→    for (const r of plan.rolls) {
   358→      const res = rollDice(r.notation, r.modifier);
   359→      const success = r.target_type === "none" ? true : res.total >= r.target;
   360→      const rr: ResolvedRoll = {
   361→        label: r.label, notation: r.notation, modifier: r.modifier,
   362→        result: res.raw, total: res.total,
   363→        target: r.target_type === "none" ? undefined : r.target,
   364→        success, purpose: r.ability || "action",
   365→      };
   366→      playerRolls.push(rr);
   367→      await logDiceRoll(roomId, round, actorName, rr);
   368→    }
   369→    outcome = playerRolls.every((r) => r.success) ? "success" : "failure";
   370→  }
   371→
   372→  const branch = outcome === "success" ? plan.success : plan.failure;
   373→
   374→  let damageDealtToMonster = 0;
   375→  let monsterThatDied: string | null = null;
   376→  let damageDealtToPlayer = 0;
   377→  let damagedPlayer: string | null = null;
   378→  let healingToPlayer = 0;
   379→  let healedPlayer: string | null = null;
   380→  const inventoryChanges: InventoryChange[] = branch.inventory || [];
   381→  let goldChange = branch.goldChange || 0;
   382→
   383→  if (branch.tokenMoves) {
   384→    for (const mv of branch.tokenMoves) {
   385→      await moveToken(roomId, mv.name, mv.newX, mv.newY, true);
   386→    }
   387→  }
   388→
   389→  if (outcome === "success" && branch.monsterDamage) {
   390→    const targetName = branch.monsterDamage.target;
   391→    let m = await db.monster.findFirst({ where: { name: { contains: targetName }, roomId, isActive: true } });
   392→    if (!m) m = await db.monster.findFirst({ where: { label: { contains: targetName }, roomId, isActive: true } });
   393→    if (!m) {
   394→      const near = await nearestActiveMonster(roomId, 0, 0);
   395→      if (near) m = await db.monster.findFirst({ where: { id: near.monster.id, roomId } });
   396→    }
   397→    if (m) {
   398→      const dmg = rollDice(branch.monsterDamage.notation);
   399→      // Talent: bonus flat damage + vampiric heal.
   400→      const bonus = damageBonusFromTalents(actorState);
   401→      const isCrit = playerRolls.some((r) => r.purpose === "СИЛ" || r.purpose === "action" || r.purpose === "ЛОВ")
   402→        ? false : false; // crit handled via natural roll below
   403→      void isCrit;
   404→      damageDealtToMonster = dmg.total + bonus;
   405→      await logDiceRoll(roomId, round, actorName, {
   406→        label: `Урон по: ${m.name}` + (bonus ? ` (+${bonus} талант)` : ""),
   407→        notation: branch.monsterDamage.notation + (bonus ? `+${bonus}` : ""),
   408→        modifier: bonus, result: dmg.raw, total: damageDealtToMonster, purpose: "player_damage",
   409→      });
   410→      const result = await damageMonster(roomId, m.id, damageDealtToMonster);
   411→      // Vampiric heal.
   412→      const vampHeal = rollVampiricHeal(actorState, damageDealtToMonster);
   413→      if (vampHeal > 0) {
   414→        await healPlayer(roomId, actorName, vampHeal);
   415→        healingToPlayer += vampHeal;
   416→        healedPlayer = actorName;
   417→        await logDiceRoll(roomId, round, actorName, {
   418→          label: "Вампиризм", notation: `${vampHeal}`, modifier: 0, result: vampHeal, total: vampHeal, purpose: "healing",
   419→        });
   420→      }
   421→      if (result.died) {
   422→        monsterThatDied = m.name;
   423→        // Heal-on-kill talent.
   424→        const killHeal = rollHealOnKill(actorState);
   425→        if (killHeal > 0) {
   426→          await healPlayer(roomId, actorName, killHeal);
   427→          healingToPlayer += killHeal;
   428→          healedPlayer = actorName;
   429→          await logDiceRoll(roomId, round, actorName, {
   430→            label: "Лечение за убийство", notation: healOnKillNotation(actorState) || `${killHeal}`, modifier: 0, result: killHeal, total: killHeal, purpose: "healing",
   431→          });
   432→        }
   433→        // Award XP to the killing player.
   434→        const xp = xpForMonster(m.maxHp);
   435→        await awardXP(roomId, actorName, xp);
   436→        await db.chatMessage.create({
   437→          data: { roomId, role: "system", speaker: "", round, content: `${actorName} получает ${xp} опыта за победу над ${m.name}.` },
   438→        });
   439→      }
   440→    }
   441→  }
   442→
   443→  if (branch.playerDamage) {
   444→    let dmg = rollDice(branch.playerDamage.notation);
   445→    let total = dmg.total;
   446→    // Talent: damage reduction.
   447→    total = applyDamageReduction(actorState, total);
   448→    damageDealtToPlayer = total;
   449→    damagedPlayer = actorName; // failure backlash hits the actor
   450→    await logDiceRoll(roomId, round, actorName, {
   451→      label: "Урон по герою" + (total < dmg.total ? ` (−${dmg.total - total} сопр.)` : ""),
   452→      notation: branch.playerDamage.notation, modifier: 0, result: dmg.raw, total, purpose: "player_damage",
   453→    });
   454→    await damagePlayer(roomId, actorName, total);
   455→  }
   456→
   457→  if (branch.healing) {
   458→    const heal = rollDice(branch.healing.notation);
   459→    healingToPlayer += heal.total;
   460→    healedPlayer = branch.healing.target || actorName;
   461→    await logDiceRoll(roomId, round, actorName, {
   462→      label: "Лечение", notation: branch.healing.notation,
   463→      modifier: 0, result: heal.raw, total: heal.total, purpose: "healing",
   464→    });
   465→    await healPlayer(roomId, healedPlayer, heal.total);
   466→  }
   467→
   468→  if (inventoryChanges.length > 0) {
   469→    await applyInventoryChanges(roomId, actorName, inventoryChanges);
   470→  }
   471→  if (goldChange) {
   472→    await adjustGold(roomId, actorName, goldChange);
   473→  }
   474→
   475→  return {
   476→    playerRolls, outcome,
   477→    damageDealtToMonster, monsterThatDied,
   478→    damageDealtToPlayer, damagedPlayer,
   479→    healingToPlayer, healedPlayer,
   480→    inventoryChanges, goldChange,
   481→    category: plan.category,
   482→    imagePrompt: plan.imagePrompt, imageNeeded: plan.imageNeeded,
   483→    branchNarrative: branch.narrative,
   484→  };
   485→}

   498→async function runMonsterTurn(roomId: string, round: number, monsterId: string): Promise<MonsterTurnResult> {
   499→  const m = await db.monster.findFirst({ where: { id: monsterId, roomId } });
   500→  if (!m) return emptyMonster();
   501→  const room = await db.room.findUnique({ where: { id: roomId } });
   502→  if (!room) return emptyMonster();
   503→
   504→  const nearest = await nearestActiveMonster(roomId, m.posX, m.posY); // not used; we want nearest player
   505→  void nearest;
   506→  const targetRes = await moveMonsterTowardNearestPlayer(roomId, m.id);
   507→  const dist = targetRes.distAfter;
   508→  const targetName = targetRes.targetName;
   509→
   510→  const rolls: ResolvedRoll[] = [];
   511→
   512→  // Not adjacent — move closer (already done), end turn.
   513→  if (dist > 1) {
   514→    return {
   515→      taken: true, rolls, damageToPlayer: 0, damagedPlayer: null,
   516→      monsterName: m.name, moved: true,
   517→      narrativeLine: `${m.name} приближается к ${targetName}.`,
   518→    };
   519→  }
   520→
   521→  // Adjacent — attack the target.
   522→  const target = await db.player.findFirst({ where: { name: targetName, roomId } });
   523→  if (!target) return emptyMonster();
   524→
   525→  // Build a PlayerState-like view for talent modifiers.
   526→  const targetState: PlayerState = {
   527→    id: target.id, name: target.name, charClass: target.charClass, level: target.level,
   528→    hp: target.hp, maxHp: target.maxHp, ac: target.ac,
   529→    str: target.str, dex: target.dex, con: target.con, int: target.int, wis: target.wis, cha: target.cha,
   530→    proficiencyBonus: target.proficiencyBonus, gold: target.gold, posX: target.posX, posY: target.posY,
   531→    color: target.color, weaponName: target.weaponName, weaponNotation: target.weaponNotation,
   532→    portraitUrl: target.portraitUrl, isHost: target.isHost, isAlive: target.isAlive,
   533→    race: target.race, raceName: target.raceName, background: target.background, backgroundName: target.backgroundName,
   534→    xp: target.xp,
   535→    selectedTalents: target.selectedTalents ? target.selectedTalents.split(",").filter(Boolean) : [],
   536→    bonusStr: target.bonusStr, bonusDex: target.bonusDex, bonusCon: target.bonusCon,
   537→    bonusInt: target.bonusInt, bonusWis: target.bonusWis, bonusCha: target.bonusCha,
   538→    pendingLevelUp: target.pendingLevelUp,
   539→  };
   540→  const targetAC = effectiveAC(targetState);
   541→
   542→  const atk = rollD20(m.attackBonus);
   543→  const hit = atk.total >= targetAC;
   544→  rolls.push({
   545→    label: `Атака ${m.name}`, notation: "1d20", modifier: m.attackBonus,
   546→    result: atk.rolls[0], total: atk.total, target: targetAC, success: hit,
   547→    purpose: "monster_attack",
   548→  });
   549→  await logDiceRoll(roomId, round, m.name, rolls[rolls.length - 1]);
   550→
   551→  if (!hit) {
   552→    return {
   553→      taken: true, rolls, damageToPlayer: 0, damagedPlayer: null,
   554→      monsterName: m.name, moved: false,
   555→      narrativeLine: `${m.name} бьёт по ${targetName}, но промахивается (${atk.total} против AC ${targetAC}).`,
   556→    };
   557→  }
   558→
   559→  const rawDmg = rollDice(m.damageNotation);
   560→  // Talent: damage reduction.
   561→  const dmg = applyDamageReduction(targetState, rawDmg.total);
   562→  await logDiceRoll(roomId, round, m.name, {
   563→    label: `Урон: ${m.name}` + (dmg < rawDmg.total ? ` (−${rawDmg.total - dmg} сопр.)` : ""),
   564→    notation: m.damageNotation, modifier: 0, result: rawDmg.raw, total: dmg, purpose: "monster_damage",
   565→  });
   566→  await damagePlayer(roomId, targetName, dmg);
   567→
   568→  // Talent: counterattack — the target may strike back.
   569→  const counterDmg = rollCounterattack(targetState);
   570→  let counterLine = "";
   571→  if (counterDmg > 0) {
   572→    const monsterResult = await damageMonster(roomId, m.id, counterDmg);
   573→    await logDiceRoll(roomId, round, targetName, {
   574→      label: `Контратака ${targetName}`, notation: "талант",
   575→      modifier: 0, result: counterDmg, total: counterDmg, purpose: "player_damage",
   576→    });
   577→    counterLine = ` ${targetName} отвечает контратакой (${counterDmg} урона)!`;
   578→    if (monsterResult.died) {
   579→      const xp = xpForMonster(m.maxHp);
   580→      await awardXP(roomId, targetName, xp);
   581→      await db.chatMessage.create({
   582→        data: { roomId, role: "system", speaker: "", round, content: `${targetName} получает ${xp} опыта за победу над ${m.name}.` },
   583→      });
   584→    }
   585→  }
   586→
   587→  return {
   588→    taken: true, rolls, damageToPlayer: dmg, damagedPlayer: targetName,
   589→    monsterName: m.name, moved: false,
   590→    narrativeLine: `${m.name} бьёт ${targetName} и попадает! ${dmg} урона (${atk.total} против AC ${targetAC}).${counterLine}`,
   591→  };
   592→}

   598→// ---------- turn advancement ----------
   599→async function advanceTurn(roomCode: string, roomId: string): Promise<{
   600→  ended: boolean;
   601→  monsterTurns: { name: string; narrative: string; result: MonsterTurnResult }[];
   602→  nextTurnName: string | null;
   603→  nextTurnType: "player" | "monster" | null;
   604→}> {
   605→  const monsterTurns: { name: string; narrative: string; result: MonsterTurnResult }[] = [];
   606→  const room = await db.room.findUnique({ where: { id: roomId } });
   607→  if (!room || !room.combatActive) {
   608→    return { ended: false, monsterTurns, nextTurnName: null, nextTurnType: null };
   609→  }
   610→
   611→  let order = await db.initiativeEntry.findMany({ where: { roomId }, orderBy: { order: "asc" } });
   612→  if (order.length === 0) {
   613→    return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
   614→  }
   615→
   616→  // Loop: advance turnIndex, run monster turns until a living player is up.
   617→  let safety = 0;
   618→  while (safety++ < 50) {
   619→    let nextIndex = room.turnIndex + 1;
   620→    let round = room.round;
   621→    if (nextIndex >= order.length) {
   622→      nextIndex = 0;
   623→      round += 1;
   624→    }
   625→    await setRoomState(roomId, { turnIndex: nextIndex, round });
   626→    // re-read room + order
   627→    const room2 = await db.room.findUnique({ where: { id: roomId } });
   628→    if (!room2) break;
   629→    room.turnIndex = room2.turnIndex;
   630→    room.round = room2.round;
   631→    order = await db.initiativeEntry.findMany({ where: { roomId }, orderBy: { order: "asc" } });
   632→    const current = order[room.turnIndex];
   633→    if (!current) {
   634→      return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
   635→    }
   636→
   637→    // Combat end checks.
   638→    const alive = await countAlive(roomId);
   639→    if (!alive.anyMonsterAlive || !alive.anyPlayerAlive) {
   640→      await setRoomState(roomId, { combatActive: false });
   641→      return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
   642→    }
   643→
   644→    if (current.combatantType === "player") {
   645→      // Skip dead players.
   646→      const p = await db.player.findFirst({ where: { name: current.combatantName, roomId } });
   647→      if (!p || p.hp <= 0 || !p.isAlive) {
   648→        // mark entry dead, continue advancing
   649→        await db.initiativeEntry.updateMany({ where: { id: current.id }, data: { isAlive: false } });
   650→        continue;
   651→      }
   652→      return { ended: false, monsterTurns, nextTurnName: current.combatantName, nextTurnType: "player" };
   653→    }
   654→
   655→    // Monster turn.
   656→    const monster = await db.monster.findFirst({ where: { name: current.combatantName, roomId } });
   657→    if (!monster || !monster.isActive || monster.hp <= 0) {
   658→      await db.initiativeEntry.updateMany({ where: { id: current.id }, data: { isAlive: false } });
   659→      continue;
   660→    }
   661→    const result = await runMonsterTurn(roomId, room.round, monster.id);
   662→    if (result.taken) {
   663→      const snap = await getSnapshot(roomCode);
   664→      const narrative = await narrateMonsterTurn(roomCode, {
   665→        monsterName: result.monsterName!,
   666→        moved: result.moved,
   667→        targetName: result.damagedPlayer ?? null,
   668→        hit: result.damageToPlayer > 0 ? true : result.narrativeLine.includes("промах") ? false : (result.moved ? null : false),
   669→        damage: result.damageToPlayer,
   670→        attackTotal: result.rolls[0]?.total ?? null,
   671→        ac: result.rolls[0]?.target ?? null,
   672→        location: snap?.location ?? "",
   673→      });
   674→      await db.chatMessage.create({
   675→        data: { roomId, role: "dm", speaker: "", round: room.round, content: narrative },
   676→      });
   677→      monsterTurns.push({ name: result.monsterName!, narrative, result });
   678→    }
   679→    // Check combat end after monster turn.
   680→    const alive2 = await countAlive(roomId);
   681→    if (!alive2.anyMonsterAlive || !alive2.anyPlayerAlive) {
   682→      await setRoomState(roomId, { combatActive: false });
   683→      return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
   684→    }
   685→  }
   686→  return { ended: false, monsterTurns, nextTurnName: null, nextTurnType: null };
   687→}

   696→/** Resolve all mechanics (plan, dice, effects, monster turns) WITHOUT the
   697→ *  final narrative. Returns everything the SSE route needs to then stream
   698→ *  the narrative token-by-token. */
   699→export async function resolvePlayerMechanics(
   700→  roomCode: string,
   701→  actorName: string,
   702→  playerAction: string
   703→): Promise<MechanicsResult> {
   704→  const room = await db.room.findUnique({ where: { code: roomCode.toUpperCase() } });
   705→  if (!room) throw new Error("Комната не найдена.");
   706→  const roomId = room.id;
   707→
   708→  const actor = await db.player.findFirst({ where: { name: actorName, roomId } });
   709→  if (!actor) throw new Error("Герой не найден в комнате.");
   710→  if (!actor.isAlive || actor.hp <= 0) throw new Error("Павший герой не может действовать.");
   711→
   712→  const wasCombatActive = room.combatActive;
   713→  const round = room.round;
   714→
   715→  // Turn enforcement.
   716→  if (wasCombatActive) {
   717→    // In combat: must be the actor's initiative turn.
   718→    const order = await db.initiativeEntry.findMany({ where: { roomId }, orderBy: { order: "asc" } });
   719→    const current = order[room.turnIndex];
   720→    if (!current || current.combatantName !== actorName || current.combatantType !== "player") {
   721→      throw new Error("Сейчас не ваш ход. Дождитесь своей инициативы.");
   722→    }
   723→  } else {
   724→    // Exploration: must be the actor's exploration turn.
   725→    const players = await db.player.findMany({ where: { roomId }, orderBy: { createdAt: "asc" } });
   726→    const alive = players.filter((p) => p.isAlive && p.hp > 0);
   727→    if (alive.length > 1) {
   728→      const current = alive[room.explorationActorIndex % alive.length];
   729→      if (!current || current.name !== actorName) {
   730→        throw new Error(`Сейчас ход: ${current?.name ?? "?"}. Дождитесь своей очереди.`);
   731→      }
   732→    }
   733→  }
   734→
   735→  // 1. Plan the mechanics first.
   736→  const plan = await planResolution(roomCode, actorName, playerAction);
   737→
   738→  // INVALID action: the DM rejected it as impossible. Do NOT consume the turn,
   739→  // do NOT apply effects — just narrate the rejection.
   740→  if (plan.category === "invalid") {
   741→    const reason = plan.invalidReason || "Это действие невозможно в данных обстоятельствах.";
   742→    await db.chatMessage.create({ data: { roomId, role: "player", speaker: actorName, round, content: playerAction } });
   743→    await db.chatMessage.create({ data: { roomId, role: "dm", speaker: "", round, content: reason } });
   744→    const snap = await getSnapshot(roomCode);
   745→    return {
   746→      actorName,
   747→      playerRolls: [],
   748→      monsterRolls: [],
   749→      outcome: "failure",
   750→      combatStarted: false,
   751→      combatEnded: false,
   752→      damageDealtToMonster: 0,
   753→      monsterThatDied: null,
   754→      damageDealtToPlayer: 0,
   755→      damagedPlayer: null,
   756→      healingToPlayer: 0,
   757→      healedPlayer: null,
   758→      inventoryChanges: [],
   759→      goldChange: 0,
   760→      imagePrompt: "",
   761→      imageNeeded: false,
   762→      branchNarrative: reason,
   763→      playerAction,
   764→      location: snap?.location ?? "",
   765→      nextTurn: wasCombatActive ? null : actorName, // turn NOT advanced for invalid actions
   766→      nextTurnType: wasCombatActive ? null : "player",
   767→      round,
   768→    };
   769→  }
   770→
   771→  // If this is the opening combat action, reveal hidden monsters so the
   772→  // attack can actually hit them.
   773→  let combatStarted = false;
   774→  if (!wasCombatActive && plan.category === "combat") {
   775→    await db.monster.updateMany({ where: { roomId, isActive: false }, data: { isActive: true } });
   776→  }
   777→
   778→  // 2. Resolve the player's action with the pre-computed plan (no monster turn).
   779→  const res = await resolvePlayerAction(roomCode, roomId, actorName, playerAction, round, plan);
   780→  let combatEnded = false;
   781→
   782→  // 3. If the action triggered combat, roll initiative (monsters already active).
   783→  if (!wasCombatActive && res.category === "combat") {
   784→    await rollInitiative(roomId);
   785→    await setRoomState(roomId, { combatActive: true, round: 1, turnIndex: 0 });
   786→    combatStarted = true;
   787→  }
   788→
   789→  // Snapshot for location (narrative will be streamed by the route).
   790→  const snap1 = await getSnapshot(roomCode);
   791→
   792→  // Persist the player's message immediately (DM narrative is saved by the route after streaming).
   793→  await db.chatMessage.create({ data: { roomId, role: "player", speaker: actorName, round, content: playerAction } });
   794→
   795→  // 4. Check immediate combat end (e.g. one-shot kill before initiative, or last monster died).
   796→  const aliveCheck = await countAlive(roomId);
   797→  if (combatStarted && (!aliveCheck.anyMonsterAlive || !aliveCheck.anyPlayerAlive)) {
   798→    await setRoomState(roomId, { combatActive: false });
   799→    combatEnded = true;
   800→  }
   801→  // Also if already in combat and the player's action ended it.
   802→  if (wasCombatActive && !aliveCheck.anyMonsterAlive) {
   803→    await setRoomState(roomId, { combatActive: false });
   804→    combatEnded = true;
   805→  }
   806→
   807→  // 5. Advance turn (run monster turns until a player is up), unless combat just ended.
   808→  let nextTurnName: string | null = null;
   809→  let nextTurnType: "player" | "monster" | null = null;
   810→  let monsterRolls: ResolvedRoll[] = [];
   811→
   812→  if (combatStarted && !combatEnded) {
   813→    // After a combat-triggering opening strike: advance from turnIndex 0.
   814→    // If order[0] is the actor (who just acted), skip to the next combatant.
   815→    const order = await db.initiativeEntry.findMany({ where: { roomId }, orderBy: { order: "asc" } });
   816→    const first = order[0];
   817→    if (first && first.combatantName === actorName && first.combatantType === "player") {
   818→      // The actor already acted (opening strike) — advance once.
   819→      const adv = await advanceTurn(roomCode, roomId);
   820→      if (adv.ended) combatEnded = true;
   821→      nextTurnName = adv.nextTurnName;
   822→      nextTurnType = adv.nextTurnType;
   823→      for (const mt of adv.monsterTurns) monsterRolls.push(...mt.result.rolls);
   824→    } else {
   825→      // Start from turnIndex 0. If it's a monster, run monster turns.
   826→      const adv = await advanceTurn(roomCode, roomId);
   827→      if (adv.ended) combatEnded = true;
   828→      nextTurnName = adv.nextTurnName;
   829→      nextTurnType = adv.nextTurnType;
   830→      for (const mt of adv.monsterTurns) monsterRolls.push(...mt.result.rolls);
   831→      // If next is a player, that's whose turn it is now.
   832→    }
   833→  } else if (wasCombatActive && !combatEnded) {
   834→    // Normal in-combat advance after the player's turn.
   835→    const adv = await advanceTurn(roomCode, roomId);
   836→    if (adv.ended) combatEnded = true;
   837→    nextTurnName = adv.nextTurnName;
   838→    nextTurnType = adv.nextTurnType;
   839→    for (const mt of adv.monsterTurns) monsterRolls.push(...mt.result.rolls);
   840→  } else if (!wasCombatActive) {
   841→    // Exploration: advance to the next alive player.
   842→    await advanceExplorationTurn(roomId, actorName);
   843→    const snap2 = await getSnapshot(roomCode);
   844→    nextTurnName = snap2?.currentExplorerName ?? null;
   845→    nextTurnType = nextTurnName ? "player" : null;
   846→  }
   847→
   848→  // Final snapshot for round reporting.
   849→  const finalRoom = await db.room.findUnique({ where: { id: roomId } });
   850→
   851→  return {
   852→    actorName: actorName,
   853→    playerRolls: res.playerRolls,
   854→    monsterRolls,
   855→    outcome: res.outcome,
   856→    combatStarted,
   857→    combatEnded,
   858→    damageDealtToMonster: res.damageDealtToMonster,
   859→    monsterThatDied: res.monsterThatDied,
   860→    damageDealtToPlayer: res.damageDealtToPlayer,
   861→    damagedPlayer: res.damagedPlayer,
   862→    healingToPlayer: res.healingToPlayer,
   863→    healedPlayer: res.healedPlayer,
