    69→/** Parse a JSON spell-slot string into a Record<string, number>. Defensive. */
    70→function parseSlotsSafe(raw: string | null | undefined): Record<string, number> {
    71→  if (!raw) return {};
    72→  try {
    73→    const parsed = JSON.parse(raw);
    74→    if (parsed && typeof parsed === "object" && !Array.isArray(parsed)) {
    75→      const out: Record<string, number> = {};
    76→      for (const [k, v] of Object.entries(parsed as Record<string, unknown>)) {
    77→        out[k] = Math.max(0, Math.floor(Number(v) || 0));
    78→      }
    79→      return out;
    80→    }
    81→  } catch {
    82→    /* ignore */
    83→  }
    84→  return {};
    85→}

    88→// ---------- Positional advantage: flanking & high ground ----------
    89→/** True if an ally of the attacker is on the opposite side of the target
    90→ *  (same row or column, equidistant, both adjacent to the target).
    91→ *  Represents D&D 5e flanking — melee only (attacker must be adjacent). */
    92→export function hasFlanking(
    93→  attacker: { posX: number; posY: number },
    94→  target: { posX: number; posY: number },
    95→  allies: { posX: number; posY: number }[]
    96→): boolean {
    97→  const dx = attacker.posX - target.posX;
    98→  const dy = attacker.posY - target.posY;
    99→  // Attacker must be adjacent to the target (Chebyshev distance 1).
   100→  if (Math.max(Math.abs(dx), Math.abs(dy)) !== 1) return false;
   101→  for (const ally of allies) {
   102→    const alx = ally.posX - target.posX;
   103→    const aly = ally.posY - target.posY;
   104→    // Ally must be adjacent to the target.
   105→    if (Math.max(Math.abs(alx), Math.abs(aly)) !== 1) continue;
   106→    // Same row (dy==0, aly==0): opposite x, equidistant.
   107→    if (dy === 0 && aly === 0 && Math.sign(alx) === -Math.sign(dx) && Math.abs(alx) === Math.abs(dx)) {
   108→      return true;
   109→    }
   110→    // Same column (dx==0, alx==0): opposite y, equidistant.
   111→    if (dx === 0 && alx === 0 && Math.sign(aly) === -Math.sign(dy) && Math.abs(aly) === Math.abs(dy)) {
   112→      return true;
   113→    }
   114→  }
   115→  return false;
   116→}

   118→/** True if the attacker is at least 3 cells higher (greater Y) than the target.
   119→ *  Ranged-only positional advantage — high ground gives advantage on attacks. */
   120→export function hasHighGround(
   121→  attacker: { posX: number; posY: number },
   122→  target: { posX: number; posY: number }
   123→): boolean {
   124→  return attacker.posY >= target.posY + 3;
   125→}
   126→
   127→/** Combined positional advantage check: flanking (melee/adjacent) OR
   128→ *  high ground (ranged/non-adjacent). Returns true if the attacker gets
   129→ *  advantage from positioning. */
   130→export function computePositionalAdvantage(
   131→  attacker: { posX: number; posY: number },
   132→  target: { posX: number; posY: number },
   133→  allies: { posX: number; posY: number }[]
   134→): boolean {
   135→  const dist = Math.max(Math.abs(attacker.posX - target.posX), Math.abs(attacker.posY - target.posY));
   136→  const isMelee = dist <= 1;
   137→  const isRanged = dist > 1;
   138→  if (isMelee && hasFlanking(attacker, target, allies)) return true;
   139→  if (isRanged && hasHighGround(attacker, target)) return true;
   140→  return false;
   141→}

   264→async function planResolution(
   265→  roomCode: string,
   266→  actorName: string,
   267→  playerAction: string
   268→): Promise<DMResolution> {
   269→  const context = await getDMContext(roomCode, actorName);
   270→  const userMsg = `КОНТЕКСТ ИГРЫ:\n${context}\n\nДЕЙСТВУЮЩИЙ ГЕРОЙ: ${actorName}\nДЕЙСТВИЕ: ${playerAction}\n\nСпланируй механику разрешения. Верни только JSON.`;
   271→  try {
   272→    const raw = await chatComplete([
   273→      { role: "system", content: SYSTEM_PROMPT_PLANNING },
   274→      { role: "user", content: userMsg },
   275→    ]);
   276→    const parsed = extractJson<DMResolution>(raw);
   277→    if (parsed && parsed.success && parsed.failure) return parsed;
   278→  } catch (e) {
   279→    console.error("[DM] planResolution error:", e);
   280→  }
   281→  return fallbackResolution(playerAction);
   282→}

   344→  playerAction: string,

   346→    playerRolls: ResolvedRoll[];

   379→  } catch (e) {
   380→    console.error("[DM] narrateAction error:", e);
   381→  }
   382→  return data.branchNarrative;

   500→async function resolvePlayerAction(
   501→  roomCode: string,
   502→  roomId: string,
   503→  actorName: string,
   504→  playerAction: string,
   505→  round: number,
   506→  plan: DMResolution
   507→): Promise<ResolutionResult> {
   508→  // Fetch the acting player's full state (for talent modifiers).
   509→  const snap0 = await getSnapshot(roomCode);
   510→  const actorState: PlayerState | undefined = snap0?.players.find((p) => p.name === actorName);
   511→  // The actor should always exist at this point (verified upstream), but
   512→  // guard against an undefined snapshot defensively so talent helpers don't
   513→  // crash — pass a no-talent stub PlayerState instead.
   514→  const actor: PlayerState = actorState ?? {
   515→    id: "", name: actorName, charClass: "", level: 1,
   516→    hp: 1, maxHp: 1, ac: 10,
   517→    str: 10, dex: 10, con: 10, int: 10, wis: 10, cha: 10,
   518→    proficiencyBonus: 2, gold: 0, posX: 0, posY: 0,
   519→    color: "#888", weaponName: "", weaponNotation: "1d4",
   520→    portraitUrl: null, isHost: false, isAlive: true,
   521→    race: "human", raceName: "Человек", background: "soldier", backgroundName: "Солдат",
   522→    xp: 0, selectedTalents: [],
   523→    bonusStr: 0, bonusDex: 0, bonusCon: 0,
   524→    bonusInt: 0, bonusWis: 0, bonusCha: 0,
   525→    pendingLevelUp: false,
   526→    spellSlots: {}, maxSpellSlots: {}, hitDice: 8,
   527→  };
   528→  const playerRolls: ResolvedRoll[] = [];
   529→  let outcome: "success" | "failure" = "success";
   530→
   531→  // Precompute advantage mode for attack rolls based on the DM plan, the
   532→  // attacker's conditions, and the target's conditions (if any).
   533→  const allConds = snap0?.conditions ?? [];
   534→  const attackerCondIds = allConds
   535→    .filter((c) => c.targetName === actorName)
   536→    .map((c) => c.condition);
   537→  // Attacker with these conditions rolls attacks at disadvantage.
   538→  const ATTACKER_DISADV_CONDS = ["poisoned", "blinded", "prone", "frightened"];
   539→  const attackerHasDisadv = attackerCondIds.some((t) => ATTACKER_DISADV_CONDS.includes(t));
   540→  // Target with these conditions gives the attacker advantage.
   541→  const TARGET_ADV_CONDS = ["blinded", "prone", "stunned"];
   542→  // Blessed condition grants +1d4 to attack rolls and saves (not advantage).
   543→  const blessedDice = attackBonusDice(attackerCondIds); // 4 for bless, 0 otherwise
   544→  // Identify the attack target up front (used for target-condition lookup).
   545→  const attackTargetName =
   546→    plan.success?.monsterDamage?.target ?? plan.failure?.monsterDamage?.target ?? null;
   547→  const targetCondIds = attackTargetName
   548→    ? allConds.filter((c) => c.targetName === attackTargetName).map((c) => c.condition)
   549→    : [];
   550→  const targetGivesAdv = targetCondIds.some((t) => TARGET_ADV_CONDS.includes(t));
   551→
   552→  // ===== Positional advantage: flanking (melee) and high ground (ranged) =====
   553→  // Look up the target monster's position (if the target is a monster).
   554→  const targetMonster = attackTargetName
   555→    ? snap0?.monsters.find((m) => m.name === attackTargetName && m.isActive) ?? null
   556→    : null;
   557→  const allies = snap0?.players.filter((p) => p.name !== actorName && p.isAlive && p.hp > 0) ?? [];
   558→  const positionalAdv = targetMonster
   559→    ? computePositionalAdvantage(actor, targetMonster, allies)
   560→    : false;
   561→
   562→  if (plan.rolls.length > 0) {
   563→    for (const r of plan.rolls) {
   564→      const isAttackRoll = r.target_type === "AC";
   565→      // Compute effective advantage mode for attack rolls only.
   566→      let adv: "advantage" | "disadvantage" | "none" = "none";
   567→      if (isAttackRoll) {
   568→        if (plan.advantage === "advantage") adv = "advantage";
   569→        else if (plan.advantage === "disadvantage") adv = "disadvantage";
   570→        if (attackerHasDisadv) {
   571→          // Disadvantage from attacker — cancels plan advantage.
   572→          adv = adv === "advantage" ? "none" : "disadvantage";
   573→        }
   574→        if (targetGivesAdv) {
   575→          // Advantage from target — cancels plan/attacker disadvantage.
   576→          adv = adv === "disadvantage" ? "none" : "advantage";
   577→        }
   578→        if (positionalAdv) {
   579→          // Flanking / high ground — cancels disadvantage, sets advantage.
   580→          adv = adv === "disadvantage" ? "none" : "advantage";
   581→        }
   582→      }
   583→
   584→      // Bonus dice (blessed = +1d4) apply to attack rolls and ability checks (saves).
   585→      let bonusDice = 0;
   586→      if (blessedDice > 0) {
   587→        bonusDice = Math.floor(Math.random() * blessedDice) + 1;
   588→      }
   589→
   590→      let result: number;
   591→      let total: number;
   592→      let allRolls: number[] | undefined;
   593→      let advantageMode: "advantage" | "disadvantage" | "none" | null = null;
   594→
   595→      if (isAttackRoll && (adv === "advantage" || adv === "disadvantage")) {
   596→        // Roll 2d20, keep higher (advantage) or lower (disadvantage).
   597→        const ar = rollD20Advantage(adv, r.modifier + bonusDice);
   598→        result = ar.result;
   599→        total = ar.total;
   600→        allRolls = ar.rolls;
   601→        advantageMode = adv;
   602→      } else {
   603→        const res = rollDice(r.notation, r.modifier);
   604→        result = res.raw;
   605→        total = res.total + bonusDice;
   606→        allRolls = bonusDice > 0 ? [res.raw] : undefined;
   607→        if (isAttackRoll) advantageMode = adv === "none" ? "none" : null;
   608→      }
   609→
   610→      const success = r.target_type === "none" ? true : total >= r.target;
   611→      const rr: ResolvedRoll = {
   612→        label: r.label,
   613→        notation: r.notation,
   614→        modifier: r.modifier + bonusDice,
   615→        result,
   616→        total,
   617→        target: r.target_type === "none" ? undefined : r.target,
   618→        success,
   619→        purpose: r.ability || "action",
   620→        advantageMode,
   621→        allRolls,
   622→      };
   623→      playerRolls.push(rr);
   624→      await logDiceRoll(roomId, round, actorName, rr);
   625→    }
   626→    outcome = playerRolls.every((r) => r.success) ? "success" : "failure";
   627→  }
   628→
   629→  const branch = outcome === "success" ? plan.success : plan.failure;
   630→
   631→  let damageDealtToMonster = 0;
   632→  let monsterThatDied: string | null = null;
   633→  let damageDealtToPlayer = 0;
   634→  let damagedPlayer: string | null = null;
   635→  let healingToPlayer = 0;
   636→  let healedPlayer: string | null = null;
   637→  const inventoryChanges: InventoryChange[] = branch.inventory || [];
   638→  let goldChange = branch.goldChange || 0;
   639→
   640→  if (branch.tokenMoves) {
   641→    for (const mv of branch.tokenMoves) {
   642→      await moveToken(roomId, mv.name, mv.newX, mv.newY, true);
   643→    }
   644→  }
   645→
   646→  // ===== AoE resolution (circle / cone / line) =====
   647→  // When the DM planned an area-of-effect spell, compute affected cells and
   648→  // apply damage to every monster AND player (except the caster) inside them.
   649→  // Each target rolls a saving throw: success = half damage, fail = full.
   650→  let aoeResult: ResolutionResult["aoe"] = undefined;
   651→  if (
   652→    plan.aoeShape &&
   653→    branch.monsterDamage?.notation &&
   654→    plan.aoeOrigin &&
   655→    typeof plan.aoeSize === "number"
   656→  ) {
   657→    const shape = plan.aoeShape;
   658→    const aoeSize = Math.max(1, Math.min(8, plan.aoeSize));
   659→    const origin = plan.aoeOrigin;
   660→    const direction = plan.aoeDirection;
   661→    const cells = computeAoECells(shape, aoeSize, origin, direction);
   662→    const element = plan.aoeElement ?? "force";
   663→    const saveDC = plan.saveDC ?? 12;
   664→    const saveAbility = plan.saveAbility ?? "ТЕЛ";
   665→    aoeResult = { shape, size: aoeSize, origin, cells, element, saveDC, saveAbility };
   666→
   667→    // Gather all combatants in the affected cells (exclude the caster).
   668→    const cellSet = new Set(cells.map((c) => `${c.x},${c.y}`));
   669→    const allMonsters = await db.monster.findMany({ where: { roomId, isActive: true } });
   670→    const allPlayers = await db.player.findMany({ where: { roomId } });
   671→    const inAreaMonsters = allMonsters.filter((m) => cellSet.has(`${m.posX},${m.posY}`));
   672→    const inAreaPlayers = allPlayers.filter(
   673→      (p) => p.name !== actorName && p.hp > 0 && p.isAlive && cellSet.has(`${p.posX},${p.posY}`)
   674→    );
   675→
   676→    const damageNotation = branch.monsterDamage.notation;
   677→    // Roll the spell damage once (the same base roll applies to all targets;
   678→    // each target's save determines full vs half). Per D&D 5e, damage is
   679→    // rolled once for the whole spell.
   680→    const baseDmgRoll = rollDice(damageNotation);
   681→    const baseDamage = baseDmgRoll.total;
   682→
   683→    await logDiceRoll(roomId, round, actorName, {
   684→      label: `Урон заклинания (${element})`,
   685→      notation: damageNotation,
   686→      modifier: 0,
   687→      result: baseDmgRoll.raw,
   688→      total: baseDamage,
   689→      purpose: "player_damage",
   690→    });
   691→
   692→    // Helper: save bonus for a target based on the save ability.
   693→    const saveBonusFor = (
   694→      ability: string,
   695→      target: { str: number; dex: number; con: number; int: number; wis: number; cha: number }
   696→    ): number => {
   697→      switch (ability) {
   698→        case "СИЛ": return abilityModifier(target.str);
   699→        case "ЛОВ": return abilityModifier(target.dex);
   700→        case "ТЕЛ": return abilityModifier(target.con);
   701→        case "ИНТ": return abilityModifier(target.int);
   702→        case "МУД": return abilityModifier(target.wis);
   703→        case "ХАР": return abilityModifier(target.cha);
   704→        default: return 0;
   705→      }
   706→    };
   707→
   708→    const aoeLog: string[] = [];
   709→
   710→    // Monsters in area.
   711→    for (const m of inAreaMonsters) {
   712→      const saveBonus = 0; // monsters in this engine have no ability scores; flat +0.
   713→      const saveRoll = rollD20(saveBonus);
   714→      const saved = saveRoll.total >= saveDC;
   715→      const dmg = saved ? Math.floor(baseDamage / 2) : baseDamage;
   716→      const rr: ResolvedRoll = {
   717→        label: `Спасбросок ${m.name} (${saveAbility})`,
   718→        notation: "1d20",
   719→        modifier: saveBonus,
   720→        result: saveRoll.rolls[0],
   721→        total: saveRoll.total,
   722→        target: saveDC,
   723→        success: saved,
   724→        purpose: "monster_save",
   725→      };
   726→      playerRolls.push(rr);
   727→      await logDiceRoll(roomId, round, m.name, rr);
   728→      if (dmg > 0) {
   729→        const r = await damageMonster(roomId, m.id, dmg);
   730→        await logDiceRoll(roomId, round, actorName, {
   731→          label: `Урон по ${m.name}${saved ? " (половина, спас)" : ""}`,
   732→          notation: damageNotation,
   733→          modifier: 0,
   734→          result: dmg,
   735→          total: dmg,
   736→          purpose: "player_damage",
   737→        });
   738→        damageDealtToMonster += dmg;
   739→        if (r.died) {
   740→          if (!monsterThatDied) monsterThatDied = m.name;
   741→          const xp = xpForMonster(m.maxHp);
   742→          await awardXP(roomId, actorName, xp);
   743→          aoeLog.push(`${m.name} повержен! (+${xp} опыта)`);
   744→        } else {
   745→          aoeLog.push(`${m.name}: ${dmg} урона${saved ? " (спас, половина)" : ""}.`);
   746→        }
   747→      }
   748→    }
   749→
   750→    // Players in area (allies caught in the blast).
   751→    for (const p of inAreaPlayers) {
   752→      const saveBonus = saveBonusFor(saveAbility, p);
   753→      const saveRoll = rollD20(saveBonus);
   754→      const saved = saveRoll.total >= saveDC;
   755→      const dmg = saved ? Math.floor(baseDamage / 2) : baseDamage;
   756→      const rr: ResolvedRoll = {
   757→        label: `Спасбросок ${p.name} (${saveAbility})`,
   758→        notation: "1d20",
   759→        modifier: saveBonus,
   760→        result: saveRoll.rolls[0],
   761→        total: saveRoll.total,
   762→        target: saveDC,
   763→        success: saved,
   764→        purpose: "player_save",
   765→      };
   766→      playerRolls.push(rr);
   767→      await logDiceRoll(roomId, round, p.name, rr);
   768→      if (dmg > 0) {
   769→        const r = await damagePlayer(roomId, p.name, dmg);
   770→        await logDiceRoll(roomId, round, actorName, {
   771→          label: `Урон по ${p.name}${saved ? " (половина, спас)" : ""}`,
   772→          notation: damageNotation,
   773→          modifier: 0,
   774→          result: dmg,
   775→          total: dmg,
   776→          purpose: "player_damage",
   777→        });
   778→        damageDealtToPlayer += dmg;
   779→        if (!damagedPlayer) damagedPlayer = p.name;
   780→        if (r.died) {
   781→          aoeLog.push(`${p.name} пал в зоне заклинания!`);
   782→        } else {
   783→          aoeLog.push(`${p.name}: ${dmg} урона${saved ? " (спас, половина)" : ""}.`);
   784→        }
   785→      }
   786→    }
   787→
   788→    if (aoeLog.length > 0) {
   789→      await db.chatMessage.create({
   790→        data: {
   791→          roomId,
   792→          role: "system",
   793→          speaker: "",
   794→          round,
   795→          content: `Область заклинания (${shape}, ${element}): ${aoeLog.join(" ")}`,
   796→        },
   797→      });
   798→    }
   799→  } else if (outcome === "success" && branch.monsterDamage) {
   800→    // Single-target damage (non-AoE).
   801→    const targetName = branch.monsterDamage.target;
   802→    let m = await db.monster.findFirst({ where: { name: { contains: targetName }, roomId, isActive: true } });
   803→    if (!m) m = await db.monster.findFirst({ where: { label: { contains: targetName }, roomId, isActive: true } });
   804→    if (!m) {
   805→      const near = await nearestActiveMonster(roomId, 0, 0);
   806→      if (near) m = await db.monster.findFirst({ where: { id: near.monster.id, roomId } });
   807→    }
   808→    if (m) {
   809→      const dmg = rollDice(branch.monsterDamage.notation);
   810→      // Talent: bonus flat damage + vampiric heal.
   811→      const bonus = damageBonusFromTalents(actor);
   812→      const isCrit = playerRolls.some((r) => r.purpose === "СИЛ" || r.purpose === "action" || r.purpose === "ЛОВ")
   813→        ? false : false; // crit handled via natural roll below
   814→      void isCrit;
   815→      damageDealtToMonster = dmg.total + bonus;
   816→      await logDiceRoll(roomId, round, actorName, {
   817→        label: `Урон по: ${m.name}` + (bonus ? ` (+${bonus} талант)` : ""),
   818→        notation: branch.monsterDamage.notation + (bonus ? `+${bonus}` : ""),
   819→        modifier: bonus, result: dmg.raw, total: damageDealtToMonster, purpose: "player_damage",
   820→      });
   821→      const result = await damageMonster(roomId, m.id, damageDealtToMonster);
   822→      // Vampiric heal.
   823→      const vampHeal = rollVampiricHeal(actor, damageDealtToMonster);
   824→      if (vampHeal > 0) {
   825→        await healPlayer(roomId, actorName, vampHeal);
   826→        healingToPlayer += vampHeal;
   827→        healedPlayer = actorName;
   828→        await logDiceRoll(roomId, round, actorName, {
   829→          label: "Вампиризм", notation: `${vampHeal}`, modifier: 0, result: vampHeal, total: vampHeal, purpose: "healing",
   830→        });
   831→      }
   832→      if (result.died) {
   833→        monsterThatDied = m.name;
   834→        // Heal-on-kill talent.
   835→        const killHeal = rollHealOnKill(actor);
   836→        if (killHeal > 0) {
   837→          await healPlayer(roomId, actorName, killHeal);
   838→          healingToPlayer += killHeal;
   839→          healedPlayer = actorName;
   840→          await logDiceRoll(roomId, round, actorName, {
   841→            label: "Лечение за убийство", notation: healOnKillNotation(actor) || `${killHeal}`, modifier: 0, result: killHeal, total: killHeal, purpose: "healing",
   842→          });
   843→        }
   844→        // Award XP to the killing player.
   845→        const xp = xpForMonster(m.maxHp);
   846→        await awardXP(roomId, actorName, xp);
   847→        await db.chatMessage.create({
   848→          data: { roomId, role: "system", speaker: "", round, content: `${actorName} получает ${xp} опыта за победу над ${m.name}.` },
   849→        });
   850→      }
   851→    }
   852→  }
   853→
   854→  if (branch.playerDamage) {
   855→    let dmg = rollDice(branch.playerDamage.notation);
   856→    let total = dmg.total;
   857→    // Talent: damage reduction.
   858→    total = applyDamageReduction(actor, total);
   859→    damageDealtToPlayer = total;
   860→    damagedPlayer = actorName; // failure backlash hits the actor
   861→    await logDiceRoll(roomId, round, actorName, {
   862→      label: "Урон по герою" + (total < dmg.total ? ` (−${dmg.total - total} сопр.)` : ""),
   863→      notation: branch.playerDamage.notation, modifier: 0, result: dmg.raw, total, purpose: "player_damage",
   864→    });
   865→    await damagePlayer(roomId, actorName, total);
   866→  }
   867→
   868→  if (branch.healing) {
   869→    const heal = rollDice(branch.healing.notation);
   870→    healingToPlayer += heal.total;
   871→    healedPlayer = branch.healing.target || actorName;
   872→    await logDiceRoll(roomId, round, actorName, {
   873→      label: "Лечение", notation: branch.healing.notation,
   874→      modifier: 0, result: heal.raw, total: heal.total, purpose: "healing",
   875→    });
   876→    await healPlayer(roomId, healedPlayer, heal.total);
   877→  }
   878→
   879→  if (inventoryChanges.length > 0) {
   880→    await applyInventoryChanges(roomId, actorName, inventoryChanges);
   881→  }
   882→  if (goldChange) {
   883→    await adjustGold(roomId, actorName, goldChange);
   884→  }
   885→
   886→  // Apply conditions the DM planned for this outcome (success or failure).
   887→  // Determine target type (player/monster) and write a system chat line.
   888→  const plannedConditions: PlannedCondition[] = Array.isArray(branch.conditions) ? branch.conditions : [];
   889→  let appliedConditionCount = 0;
   890→  if (plannedConditions.length > 0) {
   891→    const players = await db.player.findMany({ where: { roomId }, select: { name: true } });
   892→    const monsters = await db.monster.findMany({ where: { roomId }, select: { name: true } });
   893→    const playerNames = new Set(players.map((p) => p.name));
   894→    const monsterNames = new Set(monsters.map((m) => m.name));
   895→    for (const pc of plannedConditions) {
   896→      if (!pc || !pc.target || !pc.type) continue;
   897→      const targetType: "player" | "monster" = playerNames.has(pc.target) ? "player"
   898→        : monsterNames.has(pc.target) ? "monster"
   899→        : pc.target === actorName ? "player"
   900→        : "monster";
   901→      const applied = await applyCondition(roomId, pc.target, targetType, pc.type, pc.duration ?? 3, pc.source ?? actorName);
   902→      if (applied) {
   903→        appliedConditionCount++;
   904→        const def = getCondition(applied.condition);
   905→        const nameRu = def?.name ?? applied.condition;
   906→        const icon = def?.icon ?? "❓";
   907→        await db.chatMessage.create({
   908→          data: {
   909→            roomId, role: "system", speaker: "", round,
   910→            content: `${pc.target} получает состояние: ${icon} ${nameRu} (${applied.duration} раундов).`,
   911→          },
   912→        });
   913→      }
   914→    }
   915→  }
   916→
   917→  return {
   918→    playerRolls, outcome,
   919→    damageDealtToMonster, monsterThatDied,
   920→    damageDealtToPlayer, damagedPlayer,
   921→    healingToPlayer, healedPlayer,
   922→    inventoryChanges, goldChange,
   923→    category: plan.category,
   924→    imagePrompt: plan.imagePrompt, imageNeeded: plan.imageNeeded,
   925→    branchNarrative: branch.narrative,
   926→    appliedConditionCount,
   927→    aoe: aoeResult,
   928→  };
   929→}

   942→async function runMonsterTurn(roomId: string, round: number, monsterId: string): Promise<MonsterTurnResult> {
   943→  const m = await db.monster.findFirst({ where: { id: monsterId, roomId } });
   944→  if (!m) return emptyMonster();
   945→  const room = await db.room.findUnique({ where: { id: roomId } });
   946→  if (!room) return emptyMonster();
   947→
   948→  const nearest = await nearestActiveMonster(roomId, m.posX, m.posY); // not used; we want nearest player
   949→  void nearest;
   950→  const targetRes = await moveMonsterTowardNearestPlayer(roomId, m.id);
   951→  const dist = targetRes.distAfter;
   952→  const targetName: string = targetRes.targetName ?? "";
   953→
   954→  const rolls: ResolvedRoll[] = [];
   955→
   956→  // Not adjacent — move closer (already done), end turn.
   957→  if (dist > 1) {
   958→    return {
   959→      taken: true, rolls, damageToPlayer: 0, damagedPlayer: null,
   960→      monsterName: m.name, moved: true,
   961→      narrativeLine: `${m.name} приближается к ${targetName || "героям"}.`,
   962→    };
   963→  }
   964→
   965→  // Adjacent — attack the target.
   966→  const target = await db.player.findFirst({ where: { name: targetName, roomId } });
   967→  if (!target) return emptyMonster();
   968→
   969→  // Build a PlayerState-like view for talent modifiers.
   970→  const targetState: PlayerState = {
   971→    id: target.id, name: target.name, charClass: target.charClass, level: target.level,
   972→    hp: target.hp, maxHp: target.maxHp, ac: target.ac,
   973→    str: target.str, dex: target.dex, con: target.con, int: target.int, wis: target.wis, cha: target.cha,
   974→    proficiencyBonus: target.proficiencyBonus, gold: target.gold, posX: target.posX, posY: target.posY,
   975→    color: target.color, weaponName: target.weaponName, weaponNotation: target.weaponNotation,
   976→    portraitUrl: target.portraitUrl, isHost: target.isHost, isAlive: target.isAlive,
   977→    race: target.race, raceName: target.raceName, background: target.background, backgroundName: target.backgroundName,
   978→    xp: target.xp,
   979→    selectedTalents: target.selectedTalents ? target.selectedTalents.split(",").filter(Boolean) : [],
   980→    bonusStr: target.bonusStr, bonusDex: target.bonusDex, bonusCon: target.bonusCon,
   981→    bonusInt: target.bonusInt, bonusWis: target.bonusWis, bonusCha: target.bonusCha,
   982→    pendingLevelUp: target.pendingLevelUp,
   983→    spellSlots: parseSlotsSafe(target.spellSlots),
   984→    maxSpellSlots: parseSlotsSafe(target.maxSpellSlots),
   985→    hitDice: target.hitDice ?? 8,
   986→  };
   987→  const targetAC = effectiveAC(targetState);
   988→
   989→  const atk = rollD20(m.attackBonus);
   990→  const hit = atk.total >= targetAC;
   991→  rolls.push({
   992→    label: `Атака ${m.name}`, notation: "1d20", modifier: m.attackBonus,
   993→    result: atk.rolls[0], total: atk.total, target: targetAC, success: hit,
   994→    purpose: "monster_attack",
   995→  });
   996→  await logDiceRoll(roomId, round, m.name, rolls[rolls.length - 1]);
   997→
   998→  if (!hit) {
   999→    return {
  1000→      taken: true, rolls, damageToPlayer: 0, damagedPlayer: null,
  1001→      monsterName: m.name, moved: false,
  1002→      narrativeLine: `${m.name} бьёт по ${targetName}, но промахивается (${atk.total} против AC ${targetAC}).`,
  1003→    };
  1004→  }
  1005→
  1006→  const rawDmg = rollDice(m.damageNotation);
  1007→  // Talent: damage reduction.
  1008→  const dmg = applyDamageReduction(targetState, rawDmg.total);
  1009→  await logDiceRoll(roomId, round, m.name, {
  1010→    label: `Урон: ${m.name}` + (dmg < rawDmg.total ? ` (−${rawDmg.total - dmg} сопр.)` : ""),
  1011→    notation: m.damageNotation, modifier: 0, result: rawDmg.raw, total: dmg, purpose: "monster_damage",
  1012→  });
  1013→  await damagePlayer(roomId, targetName, dmg);
  1014→
  1015→  // Talent: counterattack — the target may strike back.
  1016→  const counterDmg = rollCounterattack(targetState);
  1017→  let counterLine = "";
  1018→  if (counterDmg > 0) {
  1019→    const monsterResult = await damageMonster(roomId, m.id, counterDmg);
  1020→    await logDiceRoll(roomId, round, targetName, {
  1021→      label: `Контратака ${targetName}`, notation: "талант",
  1022→      modifier: 0, result: counterDmg, total: counterDmg, purpose: "player_damage",
  1023→    });
  1024→    counterLine = ` ${targetName} отвечает контратакой (${counterDmg} урона)!`;
  1025→    if (monsterResult.died) {
  1026→      const xp = xpForMonster(m.maxHp);
  1027→      await awardXP(roomId, targetName, xp);
  1028→      await db.chatMessage.create({
  1029→        data: { roomId, role: "system", speaker: "", round, content: `${targetName} получает ${xp} опыта за победу над ${m.name}.` },
  1030→      });
  1031→    }
  1032→  }
  1033→
  1034→  return {
  1035→    taken: true, rolls, damageToPlayer: dmg, damagedPlayer: targetName,
  1036→    monsterName: m.name, moved: false,
  1037→    narrativeLine: `${m.name} бьёт ${targetName} и попадает! ${dmg} урона (${atk.total} против AC ${targetAC}).${counterLine}`,
  1038→  };
  1039→}

  1046→async function advanceTurn(roomCode: string, roomId: string): Promise<{
  1047→  ended: boolean;
  1048→  monsterTurns: { name: string; narrative: string; result: MonsterTurnResult }[];
  1049→  nextTurnName: string | null;
  1050→  nextTurnType: "player" | "monster" | null;
  1051→}> {
  1052→  const monsterTurns: { name: string; narrative: string; result: MonsterTurnResult }[] = [];
  1053→  const room = await db.room.findUnique({ where: { id: roomId } });
  1054→  if (!room || !room.combatActive) {
  1055→    return { ended: false, monsterTurns, nextTurnName: null, nextTurnType: null };
  1056→  }
  1057→
  1058→  let order = await db.initiativeEntry.findMany({ where: { roomId }, orderBy: { order: "asc" } });
  1059→  if (order.length === 0) {
  1060→    return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
  1061→  }
  1062→
  1063→  // Loop: advance turnIndex, run monster turns until a living player is up.
  1064→  let safety = 0;
  1065→  while (safety++ < 50) {
  1066→    let nextIndex = room.turnIndex + 1;
  1067→    let round = room.round;
  1068→    let roundAdvanced = false;
  1069→    if (nextIndex >= order.length) {
  1070→      nextIndex = 0;
  1071→      round += 1;
  1072→      roundAdvanced = true;
  1073→    }
  1074→    await setRoomState(roomId, { turnIndex: nextIndex, round });
  1075→    // re-read room + order
  1076→    const room2 = await db.room.findUnique({ where: { id: roomId } });
  1077→    if (!room2) break;
  1078→    room.turnIndex = room2.turnIndex;
  1079→    room.round = room2.round;
  1080→    order = await db.initiativeEntry.findMany({ where: { roomId }, orderBy: { order: "asc" } });
  1081→    // At the start of a new round, tick all conditions (decrement durations,
  1082→    // apply per-round damage like burning, remove expired).
  1083→    if (roundAdvanced) {
  1084→      const tickMessages = await tickConditions(roomId);
  1085→      for (const msg of tickMessages) {
  1086→        await db.chatMessage.create({
  1087→          data: { roomId, role: "system", speaker: "", round, content: msg },
  1088→        });
  1089→      }
  1090→    }
  1091→    const current = order[room.turnIndex];
  1092→    if (!current) {
  1093→      return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
  1094→    }
  1095→
  1096→    // Combat end checks.
  1097→    const alive = await countAlive(roomId);
  1098→    if (!alive.anyMonsterAlive || !alive.anyPlayerAlive) {
  1099→      await setRoomState(roomId, { combatActive: false });
  1100→      return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
  1101→    }
  1102→
  1103→    if (current.combatantType === "player") {
  1104→      // Skip dead players.
  1105→      const p = await db.player.findFirst({ where: { name: current.combatantName, roomId } });
  1106→      if (!p || p.hp <= 0 || !p.isAlive) {
  1107→        // mark entry dead, continue advancing
  1108→        await db.initiativeEntry.updateMany({ where: { id: current.id }, data: { isAlive: false } });
  1109→        continue;
  1110→      }
  1111→      return { ended: false, monsterTurns, nextTurnName: current.combatantName, nextTurnType: "player" };
  1112→    }
  1113→
  1114→    // Monster turn.
  1115→    const monster = await db.monster.findFirst({ where: { name: current.combatantName, roomId } });
  1116→    if (!monster || !monster.isActive || monster.hp <= 0) {
  1117→      await db.initiativeEntry.updateMany({ where: { id: current.id }, data: { isAlive: false } });
  1118→      continue;
  1119→    }
  1120→    const result = await runMonsterTurn(roomId, room.round, monster.id);
  1121→    if (result.taken) {
  1122→      const snap = await getSnapshot(roomCode);
  1123→      const narrative = await narrateMonsterTurn(roomCode, {
  1124→        monsterName: result.monsterName!,
  1125→        moved: result.moved,
  1126→        targetName: result.damagedPlayer ?? null,
  1127→        hit: result.damageToPlayer > 0 ? true : result.narrativeLine.includes("промах") ? false : (result.moved ? null : false),
  1128→        damage: result.damageToPlayer,
  1129→        attackTotal: result.rolls[0]?.total ?? null,
  1130→        ac: result.rolls[0]?.target ?? null,
  1131→        location: snap?.location ?? "",
  1132→      });
  1133→      await db.chatMessage.create({
  1134→        data: { roomId, role: "dm", speaker: "", round: room.round, content: narrative },
  1135→      });
  1136→      monsterTurns.push({ name: result.monsterName!, narrative, result });
  1137→    }
  1138→    // Check combat end after monster turn.
  1139→    const alive2 = await countAlive(roomId);
  1140→    if (!alive2.anyMonsterAlive || !alive2.anyPlayerAlive) {
  1141→      await setRoomState(roomId, { combatActive: false });
  1142→      return { ended: true, monsterTurns, nextTurnName: null, nextTurnType: null };
  1143→    }
  1144→  }
  1145→  return { ended: false, monsterTurns, nextTurnName: null, nextTurnType: null };
  1146→}

  1155→/** Resolve all mechanics (plan, dice, effects, monster turns) WITHOUT the
  1156→ *  final narrative. Returns everything the SSE route needs to then stream
  1157→ *  the narrative token-by-token. */
  1158→export async function resolvePlayerMechanics(
  1159→  roomCode: string,
  1160→  actorName: string,
  1161→  playerAction: string
  1162→): Promise<MechanicsResult> {
  1163→  const room = await db.room.findUnique({ where: { code: roomCode.toUpperCase() } });
  1164→  if (!room) throw new Error("Комната не найдена.");
  1165→  const roomId = room.id;
  1166→
  1167→  const actor = await db.player.findFirst({ where: { name: actorName, roomId } });
  1168→  if (!actor) throw new Error("Герой не найден в комнате.");
  1169→  if (!actor.isAlive || actor.hp <= 0) throw new Error("Павший герой не может действовать.");
  1170→
  1171→  const wasCombatActive = room.combatActive;
  1172→  const round = room.round;
  1173→
  1174→  // Turn enforcement.
  1175→  if (wasCombatActive) {
  1176→    // In combat: must be the actor's initiative turn.
  1177→    const order = await db.initiativeEntry.findMany({ where: { roomId }, orderBy: { order: "asc" } });
  1178→    const current = order[room.turnIndex];
  1179→    if (!current || current.combatantName !== actorName || current.combatantType !== "player") {
  1180→      throw new Error("Сейчас не ваш ход. Дождитесь своей инициативы.");
